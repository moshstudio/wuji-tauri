/** 嗅探到的单条资源 */
export interface SniffedResource {
  /** 资源 URL */
  url: string;
  /** 资源类型: "video" | "audio" | "image" | "xhr" | "fetch" | "other" */
  resourceType: string;
  /** 请求方法（XHR/Fetch 才有） */
  method?: string;
  /** Content-Type 响应头 */
  contentType?: string;
  /** 资源大小（字节） */
  size?: number;
  /** 请求发送的数据（针对 xhr/fetch） */
  requestData?: string;
  /** 响应内容（截断处理，针对 xhr/fetch） */
  responseBody?: string;
}

/**
 * 通过隐藏 Webview 窗口加载指定 URL，返回解析后的 Document 对象。
 * 同时自动嗅探页面中的视频、音频、图片及 XHR/Fetch 请求链接。
 *
 * 返回的 Document 对象额外挂载了以下属性（通过 Proxy）：
 * - `cookie`: 页面 Cookie
 * - `URL`: 页面最终 URL
 * - `location`: 页面最终 URL（URL 对象）
 * - `_sniffedResources`: 嗅探到的资源列表 (SniffedResource[])
 *
 * @param url 目标页面 URL
 * @param options 配置项
 * @param options.timeout 超时时长 (单位: 秒, 默认 20)
 * @param options.waitForResources 指定需等待嗅探到的资源类型（支持单种或多种，如 'video' 或 ['video', 'xhr']），抓取器将等待这些资源出现并稳定后再返回。
 * @param options.useSavedCookie 是否自动补全已保存的 Cookie (默认 true)
 * @returns 解析后的 Document 对象，若失败返回 null
 *
 * @example
 * // 基础用法
 * const doc = await fetchWebview('https://example.com');
 * 
 * // 爬取视频页面：等待视频和 XHR 资源请求稳定
 * const doc = await fetchWebview('https://v.qq.com/xxx', {
 *   waitForResources: ['video', 'xhr'],
 *   timeout: 30
 * });
 *
 * if (doc) {
 *   const resources = (doc as any)._sniffedResources as SniffedResource[];
 * }
 */
export declare function fetchWebview(
  url: string,
  options?: {
    timeout?: number;
    waitForResources?:
      | 'video'
      | 'audio'
      | 'image'
      | 'xhr'
      | 'fetch'
      | Array<'video' | 'audio' | 'image' | 'xhr' | 'fetch'>;
    useSavedCookie?: boolean;
  }
): Promise<Document | null>;
